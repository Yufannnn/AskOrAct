\begin{frame}{Motivation and Setup}
  \begin{columns}[T,onlytextwidth]
    \column{0.50\textwidth}
      \includegraphics[width=\textwidth]{setup_overview.png}
      \vspace{0.3em}
      \begin{itemize}\small
        \item $9{\times}9$ two-room gridworld, $K$ candidate goals
        \item Boltzmann principal ($\beta$, $\varepsilon$-noise)
        \item 6 policies, 19{,}800 episodes, 95\% bootstrap CIs
      \end{itemize}
    \column{0.48\textwidth}
      \begin{block}{The question}
        \small
        An assistant can infer the principal's goal by
        \textbf{passively watching} actions or
        \textbf{actively asking} at cost $c_q$.\\[0.4em]
        \emph{How do these two channels interact --- and when does more observation hurt?}
      \end{block}
      \vspace{0.3em}
      \begin{block}{Gap}
        \small
        Prior work (CLIPS, ATD, ShootFirst) studies each channel in isolation.
        To our knowledge this is the \alert{first benchmark combining both}
        under controlled ambiguity sweeps.
      \end{block}
  \end{columns}
\end{frame}

% =============================================================
\begin{frame}{Finding 1 \& 2 --- Redundancy and Non-Monotonicity}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \centering
      \includegraphics[width=\textwidth,height=3.6cm,keepaspectratio]{synergy_decomposition.png}\\[0.3em]
      \small\textbf{Channels are sub-additive} (in this setting)\\
      $I{=}{-}26{\pm}3$\,pp at $K{=}3$.
      Clarification matters most when passive inference is weak ---
      \alert{$+22.6$\,pp} under full first-action aliasing.
    \column{0.48\textwidth}
      \centering
      \includegraphics[width=\textwidth,height=3.6cm,keepaspectratio]{beta_inverted_u.png}\\[0.3em]
      \small\textbf{Assistability is non-monotonic in $\beta$} (in our benchmark)\\
      NeverAsk peaks at $\beta{=}8$ ($92\%$) then \alert{drops to $84\%$}
      at $\beta{=}16$ as action diversity collapses.
      AskOrAct detects this and asks more.
  \end{columns}
\end{frame}

% =============================================================
\begin{frame}{Finding 3 --- More Observation Can Hurt, and the Fix}
  \begin{columns}[T,onlytextwidth]
    \column{0.46\textwidth}
      \centering
      \includegraphics[width=\textwidth,height=3.4cm,keepaspectratio]{obs_hurts_mismatch.png}\\[0.3em]
      \small
      Forced passive observation costs \alert{$-2.2$\,pp/step} under
      $\beta$-mismatch vs.\ $-0.8$ matched.
      Three decision-layer fixes all fail --- the bottleneck is
      \alert{posterior quality}.
    \column{0.52\textwidth}
      \begin{block}{AskOrAct-ST: fix the posterior}
        \small
        Temper passive updates by $\omega_t \in (0,1]$ from a
        CUSUM surprise accumulator.
        Down-weight observations that are unlikely under the assumed model.
        \begin{itemize}\small
          \item Matched: $+0.0 \pm 1.6$\,pp (no harm)
          \item Mismatched ($\beta_{\text{true}}{=}1$): \alert{$+1$--$+5$\,pp}
        \end{itemize}
      \end{block}
      \vspace{0.4em}
      \begin{block}{Three takeaways}
        \small
        \begin{enumerate}
          \item Channels are sub-additive --- exploit redundancy, don't stack.
          \item Rationality can \emph{hurt} passive inference.
          \item Under mismatch, fix the \textbf{posterior}, not the gate.
        \end{enumerate}
      \end{block}
  \end{columns}
\end{frame}
